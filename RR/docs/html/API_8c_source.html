319 int ActualizarDistancias(DovahkiinP dova){
320  Queue q, qNext; /*Colas para el manejo de los niveles.
321  q = nodos del nivel actual; qNext = del siguiente*/
322  Network *node = NULL; /*Nodo actual de 'q' en el cual se itera*/
323  Network *k = NULL; /*Iterador para reseteo de las distancias*/
324 
325  assert(dova != NULL);
326 
327  /*Preparación de las cosas que voy a usar*/
328  UNSET_FLAG(SINK_REACHED);
329  HASH_CLEAR(hhCut, dova->cut);
330  q = queue_create();
331  qNext = queue_create();
332 
333  /* Reset de distancias por llamadas anteriores*/
334  for(k = dova->net; k != NULL; k = k->hhNet.next){
335  k->lvl = LVL_NIL;
336  }
337  /*La fuente es nivel 0*/
338  HASH_FIND(hhNet, dova->net, &(dova->src), sizeof(dova->src), node);
339  assert(node != NULL);
340  node->lvl = 0;
341  queue_enqueue(q, node);
342 
343  /*Actualización de distancias por BFS */
344  while(!queue_isEmpty(q) && !IS_SET_FLAG(SINK_REACHED)){
345  node = queue_head(q);
346  /*Actualización de niveles de distancia de los nodos vecinos de 'node'*/
347  set_lvlNbrs(dova, node, qNext);
348  /*Agrego 'node' al corte*/
349  HASH_ADD(hhCut, dova->cut, name, sizeof(dova->cut->name), node);
350  queue_dequeue(q);
351  /*Si se terminaron los nodos de este nivel, se pasa al siguiente*/
352  if(queue_isEmpty(q))
353  queue_swap(&q, &qNext);
354  }
355  /*Si se alcanzo 't'*/
356  if(IS_SET_FLAG(SINK_REACHED)){
357  HASH_CLEAR(hhCut, dova->cut); /*EL corte no es minimal, no me sirve*/
358  }else
359  SET_FLAG(MAXFLOW); /*El flujo es maximal => hay corte minimal*/
360 
361  queue_destroy(q, NULL);
362  queue_destroy(qNext, NULL);
363  return IS_SET_FLAG(SINK_REACHED);

670 static Network *network_nextNode(Network *net, Network *x, short int *dir){
671  Network *y = NULL; /*El nodo candidato a ser el siguiente. Retorno*/
672  int getNbr; /*Resultado de los nbrhd_getX()*/
673  u64 yName; /*Nombre del nodo 'y'*/
674  u64 flow, cap; /*Flujo y capacidad entre 'x' e 'y'*/
675  bool breakW = false; /*Termina la búsqueda iterativa*/
676 
677  assert(net != NULL && x != NULL);
678  /*Inicio con el primer vecino que encuentre*/
679  getNbr = nbrhd_getFwd(x->nbrs, FST, &yName);
680  if(!getNbr){
681  getNbr = nbrhd_getBwd(x->nbrs, FST, &yName);
682  *dir = BWD;
683  }else
684  *dir = FWD;
685 
686  while(getNbr && !breakW){
687  HASH_FIND(hhNet, net, &yName, sizeof(yName), y);
688  assert(y!=NULL);
689  /*Compruebo si su distancia es mayor por 1 unidad*/
690  if(y->lvl != LVL_NIL && y->lvl == (x->lvl+1)){
691  flow = nbrhd_getFlow(x->nbrs, yName, *dir);
692  cap = nbrhd_getCap(x->nbrs, yName, *dir);
693  /*Compruebo el envio de flujo. True = Cumple lo requerido! salgo*/
694  breakW = (*dir == FWD && cap > flow) || (*dir == BWD && flow > 0);
695  }
696  /*Si no es valido, se descarta y busca otro (prioridad por FWD)*/
697  if(*dir == FWD && !breakW){
698  getNbr = nbrhd_getFwd(x->nbrs, NXT, &yName);
699  if(!getNbr){ /*Se acabaron los FWD, sigo con los BWD*/
700  getNbr = nbrhd_getBwd(x->nbrs, FST, &yName);
701  *dir = BWD;
702  }
703  }else if(*dir == BWD && !breakW)
704  getNbr = nbrhd_getBwd(x->nbrs, NXT, &yName);
705  }
706 
707  if(!breakW && !getNbr)
708  y = NULL; /*Ningún vecino cualifica*/
709 
710  return y;
711 }

724 static void set_lvlNbrs(DovahkiinP dova, Network *x, Queue upd){
725  Network *y = NULL; /*Nodo 'y' vecino de 'x' por actualizar*/
726  int getNbr; /*Resultado de los nbrhd_getX()*/
727  int dir = FWD; /*Dirección en la que se encuentra el vecino*/
728  u64 yName, cap, flow; /*Nombre del nodo 'y', capacidad y flujo con 'x'*/
729 
730  assert(dova != NULL);
731  assert(x != NULL && upd != NULL);
732 
733  /*Inicio con el primer vecino que encuentre*/
734  getNbr = nbrhd_getFwd(x->nbrs, FST, &yName);
735  if(!getNbr){
736  getNbr = nbrhd_getBwd(x->nbrs, FST, &yName);
737  dir = BWD;
738  }
739  /*No paro hasta que alcance 't' o se me acaben los vecinos*/
740  while(getNbr && !IS_SET_FLAG(SINK_REACHED)){
741  flow = nbrhd_getFlow(x->nbrs, yName, dir);
742  cap = nbrhd_getCap(x->nbrs, yName, dir);
743  /*Según su dirección, compruebo el envío de flujo*/
744  if((dir == FWD && cap > flow) || (dir == BWD && flow > 0)){
745  /*Busco el nodo y actualizo su nivel (si todavía no lo hizo)*/
746  HASH_FIND(hhNet, dova->net, &yName, sizeof(yName), y);
747  assert(y != NULL);
748  if(y->lvl == LVL_NIL){
749  y->lvl = x->lvl + 1;
750  queue_enqueue(upd, y);
751  if(yName == dova->snk) /*Se llego a t*/
752  SET_FLAG(SINK_REACHED);
753  }
754  }
755  /*Busco el siguiente vecino*/
756  if(dir == FWD && !IS_SET_FLAG(SINK_REACHED)){
757  getNbr = nbrhd_getFwd(x->nbrs, NXT, &yName);
758  if(!getNbr){ /*Se acabaron los FWD, sigo con los BWD*/
759  getNbr = nbrhd_getBwd(x->nbrs, FST, &yName);
